248 int function_signatures::num_promotions(const std::vector<expr_type>& call_args, const std::vector<expr_type>& sig_args)
249 {
250  if (call_args.size() != sig_args.size()) {
251  return -1; // failure
252  }
253  int num_promotions = 0;
254  for (size_t i = 0; i < call_args.size(); ++i) {
255  if (call_args[i] == sig_args[i]) {
256  continue;
257  } else if (call_args[i].is_primitive_int() && sig_args[i].is_primitive_double()) {
258  ++num_promotions;
259  } else {
260  return -1; // failed match
261  }
262  }
263  return num_promotions;
264 }

266 expr_type function_signatures::get_result_type(const std::string & name,
267  const std::vector<expr_type>& args,
268  std::ostream & error_msgs)
269 {
270  std::vector<function_signature_t> signatures = sigs_map_[name];
271  size_t match_index = 0;
272  size_t min_promotions = std::numeric_limits<size_t>::max();
273  size_t num_matches = 0;
274 
275  for (size_t i = 0; i < signatures.size(); ++i) {
276  int promotions = num_promotions(args, signatures[i].second);
277  if (promotions < 0) {
278  continue; // no match
279  }
280  size_t promotions_ui = static_cast<size_t>(promotions);
281  if (promotions_ui < min_promotions) {
282  min_promotions = promotions_ui;
283  match_index = i;
284  num_matches = 1;
285  } else if (promotions_ui == min_promotions) {
286  ++num_matches;
287  }
288  }
289 
290  if (num_matches == 1) {
291  return signatures[match_index].first;
292  } else if (num_matches == 0) {
293  error_msgs << "no matches for function name=\"" << name << "\"" << std::endl;
294  } else {
295  error_msgs << num_matches << " matches with " << min_promotions << " integer promotions " <<
296  "for function name=\"" << name << "\"" << std::endl;
297  }
298  for (size_t i = 0; i < args.size(); ++i) {
299  error_msgs << " arg " << i << " type=" << args[i] << std::endl;
300  }
301  return expr_type(); // ill-formed dummy
302 }

594 {
