270  // a map from the source ID to a pair: <source vertex, vector of edges from that vertex to this one>
271  std::map<unsigned int, std::pair<boost::graph_traits<Graph>::vertex_descriptor, std::vector<boost::graph_traits<Graph>::in_edge_iterator> > > sources;
272 
273  // the input edges into vOld
274  boost::graph_traits<Graph>::in_edge_iterator e, e_end;
275 
276  for( tie(e, e_end)=in_edges(vOld, *graph); e!=e_end; ++e ) { // loop through the input edges
277  auto v = source(*e, *graph);
278  const unsigned int id = graph->operator[](v)->piece->ID();
279 
280  auto it = sources.find(id);
281  if( it==sources.end() ){
282  sources[id] = std::pair<boost::graph_traits<Graph>::vertex_descriptor, std::vector<boost::graph_traits<Graph>::in_edge_iterator> >(v, std::vector<boost::graph_traits<Graph>::in_edge_iterator>(1, e));
283  } else {
284  sources[id].second.push_back(e);
285  }
286  }
287 
288  // loop through the sources
289  for( auto it : sources ) {
290  // the upstream node iterator
291  auto v = it.second.first;
292 
293  if( Constant(v) && std::dynamic_pointer_cast<ConstantPiece>(graph->operator[](v)->piece)==nullptr ) { // if this node is constant but is not already a muq::Modeling::ConstantPiece
294  // get the output values for this node
295  const std::vector<boost::any>& outputs = GetConstantOutputs(v);
296 
297  // create a ConstantPiece node for this input
298  auto nextV = boost::add_vertex(*(newGraph->graph));
299  newGraph->graph->operator[](nextV) = std::make_shared<WorkGraphNode>(std::make_shared<ConstantPiece>(outputs), graph->operator[](v)->name+"_fixed");
300 
301  // loop through the edges from the source to this node
302  for( auto e : it.second.second ) {
303  if( !newGraph->HasEdge(nextV, vNew, graph->operator[](*e)->inputDim) ) { // if edge does not exist ...
304  // ... add the edge from this node to the existing node
305  auto nextE = boost::add_edge(nextV, vNew, *(newGraph->graph));
306  newGraph->graph->operator[](nextE.first) = std::make_shared<WorkGraphEdge>(graph->operator[](*e)->outputDim, graph->operator[](*e)->inputDim);
307  }
308  }
309 
310  // move to the next source
311  continue;
312  }
313 
314  // loop through the edges from the (nonconstant) source to this node
315  for( auto e : it.second.second ) {
316  boost::graph_traits<Graph>::vertex_descriptor nextV;
317  boost::graph_traits<Graph>::vertex_iterator ind;
318 
319  if( newGraph->HasNode(ind, graph->operator[](v)->name) ) { // if the node already exists in the new graph ...
320  // ... the next node is that node
321  nextV = *ind;
322  } else { // if not ...
323  // ... copy the source node over to the new graph and make an edge
324  nextV = boost::add_vertex(*(newGraph->graph));
325  newGraph->graph->operator[](nextV) = graph->operator[](v);
326  }
327 
328  if( !newGraph->HasEdge(nextV, vNew, graph->operator[](*e)->inputDim ) ) { // if the edge does not already exist ...
329  // add the edge from this node to the existing node,
330  auto nextE = boost::add_edge(nextV, vNew, (*newGraph->graph));
331  newGraph->graph->operator[](nextE.first) = std::make_shared<WorkGraphEdge>(graph->operator[](*e)->outputDim, graph->operator[](*e)->inputDim);
332  }
333 
334  // recurse down a step further
335  RecursiveCut(v, nextV, newGraph);
336  }
337  }

573 std::vector<boost::any>& WorkGraph::GetConstantOutputs(boost::graph_traits<Graph>::vertex_descriptor const& node) const {
574  // make sure the node indeed cosntant
575  assert(Constant(node));
576 
577  // the WorkPiece associated with this node
578  auto work = graph->operator[](node)->piece;
579 
580  // make sure we know the number of inputs
581  assert(work->numInputs>=0);
582 
583  // if the node has no inputs
584  if( work->numInputs==0 ) {
585  work->Evaluate();
586  return work->outputs;
587  }
588 
589  // get iterators to the input edges
590  boost::graph_traits<Graph>::in_edge_iterator e, e_end;
591 
592  // a map from the WorkPiece ID number to a pair: <upstream vertex descriptor, vector of pairs: <ouput supply, input supplied> >
593  std::map<unsigned int, std::pair<boost::graph_traits<Graph>::vertex_descriptor, std::vector<std::pair<unsigned int, unsigned int> > > > inMap;
594 
595  // loop through the input edges
596  for( tie(e, e_end)=in_edges(node, *graph); e!=e_end; ++e ) {
597  // get the WorkPiece id number, the output that it supplies, and the input that receives it
598  const unsigned int id = graph->operator[](source(*e, *graph))->piece->ID();
599  const unsigned int inNum = graph->operator[](*e)->inputDim;
600  const unsigned int outNum = graph->operator[](*e)->outputDim;
601 
602  // add to the list of inputs supplyed by the WorkPiece with this id
603  auto it = inMap.find(id);
604  if( it==inMap.end() ) {
605  inMap[id] = std::pair<boost::graph_traits<Graph>::vertex_descriptor, std::vector<std::pair<unsigned int, unsigned int> > >(boost::source(*e, *graph), std::vector<std::pair<unsigned int, unsigned int> >(1, std::pair<unsigned int, unsigned int>(outNum, inNum)));
606  } else {
607  inMap[id].second.push_back(std::pair<unsigned int, unsigned int>(outNum, inNum));
608  }
609  }
610 
611  // the inputs to this WorkPiece
612  boost::any empty(nullptr);
613  ref_vector<boost::any> ins(work->numInputs, std::cref(empty));
614 
615  // loop through the edges again, now we know which outputs supply which inputs
616  for( auto it : inMap ) {
617  // get the outputs of the upstream node
618  std::vector<boost::any>& upstreamOutputs = GetConstantOutputs(it.second.first);
619 
620  // loop through the inputs supplied by the outputs of this upstream node
621  for( auto out_in : it.second.second ) {
622  // populate the inputs to this WorkPiece
623  ins.at(out_in.second) = upstreamOutputs.at(out_in.first);
624  }
625  }
626 
627  // evaluate this node
628  work->Evaluate(ins);
629  return work->outputs;
630 }

636 bool WorkGraph::Constant(boost::graph_traits<Graph>::vertex_descriptor const& node) const {
637  // the WorkPiece associated with this node
638  auto work = graph->operator[](node)->piece;
639 
640  // if the node has no inputs, it is constant
641  if( work->numInputs==0 ) { return true; }
642 
643  // if not all of its inputs are set, it is not constant (this also covers the case that we don't know the number of inputs)
644  if( in_degree(node, *graph)!=work->numInputs ) { return false; }
645 
646  // all of the inputs are given by upstream nodes, it is constant if the upstream nodes are constant
647  bool constant = true;
648 
649  // get iterators to the input edges
650  boost::graph_traits<Graph>::in_edge_iterator e, e_end;
651 
652  // loop through the input edges
653  for( tie(e, e_end)=in_edges(node, *graph); e!=e_end; ++e ) {
654  // if an upstream node is not constant, then this node is not constant either
655  if( !Constant(boost::source(*e, *graph)) ) { return false; }
656  }
657 
658  // all upstream nodes are constant, so this node is alos constant
659  return true;
660 }
