9 int calculate_size(const Eigen::VectorXd& x, const std::string& name, const int digits, std::ios_base::fmtflags& format)
10 {
11  using std::max;
12  using std::ceil;
13  using std::log10;
14 
15  double padding = 0;
16  if (digits > 0) {
17  padding = digits + 1;
18  }
19 
20  double fixed_size = 0.0;
21  if (x.maxCoeff() > 0) {
22  fixed_size = ceil(log10(x.maxCoeff() + 0.001)) + padding;
23  }
24  if (x.minCoeff() < 0) {
25  fixed_size = max(fixed_size, ceil(log10(-x.minCoeff() + 0.01)) + (padding + 1));
26  }
27  format = std::ios_base::fixed;
28  if (fixed_size < 7) {
29  return max(fixed_size, max(name.length(), std::string("-0.0").length()) + 0.0);
30  }
31 
32  double scientific_size = 0;
33  scientific_size += 4.0; // "-0.0" has four digits
34  scientific_size += 1.0; // e
35  double exponent_size = 0;
36  if (x.maxCoeff() > 0) {
37  exponent_size = ceil(log10(log10(x.maxCoeff())));
38  }
39  if (x.minCoeff() < 0) {
40  exponent_size = max(exponent_size, ceil(log10(log10(-x.minCoeff()))));
41  }
42  scientific_size += fmin(exponent_size, 3);
43  format = std::ios_base::scientific;
44  return scientific_size;
45 }
