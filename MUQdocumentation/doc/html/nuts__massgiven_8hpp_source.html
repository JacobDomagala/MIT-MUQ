160  // Initialize the algorithm
161  std::vector<double> mminus(this->_model.num_params_r());
162  for (size_t i = 0; i < mminus.size(); ++i)
163  mminus[i] = this->_rand_unit_norm();
164  std::vector<double> mplus(mminus);
165  // The log-joint probability of the momentum and position terms, i.e.
166  // -(kinetic energy + potential energy)

174  // Sample the slice variable

176  int nvalid = 1;
177  int direction = 2 * (this->_rand_uniform_01() > 0.5) - 1;
178  bool criterion = true;
179 
180  // Repeatedly double the set of points we've visited

182  double newlogp = -1;

192  double low = epsilon * (1.0 - this->_epsilon_pm);
193  double high = epsilon * (1.0 + this->_epsilon_pm);
194  double range = high - low;
