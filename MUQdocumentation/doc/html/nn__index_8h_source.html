288  virtual int knnSearch(const Matrix<ElementType>& queries,
289  Matrix<size_t> & indices,
290  Matrix<DistanceType> & dists,
291  size_t knn,
292  const SearchParams & params) const
293  {
294  assert(queries.cols == veclen());
295  assert(indices.rows >= queries.rows);
296  assert(dists.rows >= queries.rows);
297  assert(indices.cols >= knn);
298  assert(dists.cols >= knn);
299  bool use_heap;
300 
301  if (params.use_heap == FLANN_Undefined) {
302  use_heap = (knn > KNN_HEAP_THRESHOLD) ? true : false;
303  } else {
304  use_heap = (params.use_heap == FLANN_True) ? true : false;
305  }
306  int count = 0;
307 
308  if (use_heap) {
309 #pragma omp parallel num_threads(params.cores)
310  {
311  KNNResultSet2<DistanceType> resultSet(knn);
312 #pragma omp for schedule(static) reduction(+:count)
313  for (int i = 0; i < (int)queries.rows; i++) {
314  resultSet.clear();
315  findNeighbors(resultSet, queries[i], params);
316  size_t n = std::min(resultSet.size(), knn);
317  resultSet.copy(indices[i], dists[i], n, params.sorted);
318  indices_to_ids(indices[i], indices[i], n);
319  count += n;
320  }
321  }
322  } else {
323 #pragma omp parallel num_threads(params.cores)
324  {
325  KNNSimpleResultSet<DistanceType> resultSet(knn);
326 #pragma omp for schedule(static) reduction(+:count)
327  for (int i = 0; i < (int)queries.rows; i++) {
328  resultSet.clear();
329  findNeighbors(resultSet, queries[i], params);
330  size_t n = std::min(resultSet.size(), knn);
331  resultSet.copy(indices[i], dists[i], n, params.sorted);
332  indices_to_ids(indices[i], indices[i], n);
333  count += n;
334  }
335  }
336  }
337  return count;
338  }

472  int radiusSearch(const Matrix<ElementType>& queries,
473  Matrix<size_t> & indices,
474  Matrix<DistanceType> & dists,
475  float radius,
476  const SearchParams & params) const
477  {
478  assert(queries.cols == veclen());
479  int count = 0;
480  size_t num_neighbors = std::min(indices.cols, dists.cols);
481  int max_neighbors = params.max_neighbors;
482  if (max_neighbors < 0) {
483  max_neighbors = num_neighbors;
484  } else { max_neighbors = std::min(max_neighbors, (int)num_neighbors); }
485 
486  if (max_neighbors == 0) {
487 #pragma omp parallel num_threads(params.cores)
488  {
489  CountRadiusResultSet<DistanceType> resultSet(radius);
490 #pragma omp for schedule(static) reduction(+:count)
491  for (int i = 0; i < (int)queries.rows; i++) {
492  resultSet.clear();
493  findNeighbors(resultSet, queries[i], params);
494  count += resultSet.size();
495  }
496  }
497  } else {
498  // explicitly indicated to use unbounded radius result set
499  // and we know there'll be enough room for resulting indices and dists
500  if ((params.max_neighbors < 0) && (num_neighbors >= size())) {
501 #pragma omp parallel num_threads(params.cores)
502  {
503  RadiusResultSet<DistanceType> resultSet(radius);
504 #pragma omp for schedule(static) reduction(+:count)
505  for (int i = 0; i < (int)queries.rows; i++) {
506  resultSet.clear();
507  findNeighbors(resultSet, queries[i], params);
508  size_t n = resultSet.size();
509  count += n;
510  if (n > num_neighbors) {
511  n = num_neighbors;
512  }
513  resultSet.copy(indices[i], dists[i], n, params.sorted);
514 
515  // mark the next element in the output buffers as unused
516  if (n < indices.cols) {
517  indices[i][n] = size_t(-1);
518  }
519  if (n < dists.cols) {
520  dists[i][n] = std::numeric_limits<DistanceType>::infinity();
521  }
522  indices_to_ids(indices[i], indices[i], n);
523  }
524  }
525  } else {
526  // number of neighbors limited to max_neighbors
527 #pragma omp parallel num_threads(params.cores)
528  {
529  KNNRadiusResultSet<DistanceType> resultSet(radius, max_neighbors);
530 #pragma omp for schedule(static) reduction(+:count)
531  for (int i = 0; i < (int)queries.rows; i++) {
532  resultSet.clear();
533  findNeighbors(resultSet, queries[i], params);
534  size_t n = resultSet.size();
535  count += n;
536  if ((int)n > max_neighbors) {
537  n = max_neighbors;
538  }
539  resultSet.copy(indices[i], dists[i], n, params.sorted);
540 
541  // mark the next element in the output buffers as unused
542  if (n < indices.cols) {
543  indices[i][n] = size_t(-1);
544  }
545  if (n < dists.cols) {
546  dists[i][n] = std::numeric_limits<DistanceType>::infinity();
547  }
548  indices_to_ids(indices[i], indices[i], n);
549  }
550  }
551  }
552  }
553  return count;
554  }

700  size_t id_to_index(size_t id)
701  {
702  //begin MUQ modifications
703  if (!removed_ && (id < size_)) {
704  return id; //return the id directly if there's no removed points and it's within the size of all known
705  // samples
706  } else if (!removed_) {
707  return size_t(-1); //else id isn't part of the index, and return -1, the error code
708  }
709  size_t point_index = size_t(-1);
710 
711  if ((id < ids_.size()) && (ids_[id] == id)) { // add a size check
712  return id;
713  }
714  // if (ids_.size()==0) {
715  // return id;
716  // }
717  // size_t point_index = size_t(-1);
718  // if (ids_[id]==id) {
719  // return id;
720  // }
721  //end MUQ modifications
722  else {
723  // binary search
724  size_t start = 0;
725  size_t end = ids_.size();
726 
727  while (start < end) {
728  size_t mid = (start + end) / 2;
729  if (ids_[mid] == id) {
730  point_index = mid;
731  break;
732  } else if (ids_[mid] < id) {
733  start = mid + 1;
734  } else {
735  end = mid;
736  }
737  }
738  }
739  return point_index;
740  }

768  void extendDataset(const Matrix<ElementType>& new_points)
769  {
770  size_t new_size = size_ + new_points.rows;
771 
772  if (removed_) {
773  removed_points_.resize(new_size);

775  }

787  void cleanRemovedPoints()
788  {
789  if (!removed_) {
790  return;
791  }
792 
793  size_t last_idx = 0;
794  for (size_t i = 0; i < size_; ++i) {
795  if (!removed_points_.test(i)) {
796  points_[last_idx] = points_[i];
