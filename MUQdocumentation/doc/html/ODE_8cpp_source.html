74 void ODE::ForwardIntegration(void *cvode_mem, N_Vector& state, ref_vector<boost::any> const& outputTimes) {
75  // each element corresponds to a vector of desired times, first: the current index of that vector, second: the size of that vector
76  std::vector<std::pair<unsigned int, unsigned int> > timeIndices = TimeIndices(outputTimes);
77 
78  // first: the next time to integrate to, second: the output index
79  std::pair<double, int> nextTime;
80 
81  // start the clock at t=0
82  double tcurrent = 0.0;
83 
84  // loop through each time
85  while( NextTime(nextTime, timeIndices, outputTimes) ) {
86  // if we have to move forward --- i.e., not at the initial time or another output does not need the current time
87  if( std::fabs(nextTime.first-tcurrent)>1.0e-14 ) { // we have to move forward in time
88  int flag = CVode(cvode_mem, nextTime.first, state, &tcurrent, CV_NORMAL);
89  assert(CheckFlag(&flag, "CVode", 1));
90  }
91 
92  // save the result at this timestep
93  if( timeIndices[nextTime.second].second>1 ) { // the output has more than one compnent ...
94  // .. save the current state as an element in the vector
95  DeepCopy(boost::any_cast<std::vector<N_Vector>&>(outputs[nextTime.second]) [timeIndices[nextTime.second].first-1], state);
96  } else { // the out has one component ...
97  // ... save the current state (not inside a vector)
98  DeepCopy(boost::any_cast<N_Vector&>(outputs[nextTime.second]), state);
99  }
100  }
