41  factor_cov_matrix(Eigen::Array<T,Eigen::Dynamic,1>& CPCs,

44 
45  size_t K = sds.rows();
46 
47  sds = Sigma.diagonal().array();
48  if( (sds <= 0.0).any() ) return false;
49  sds = sds.sqrt();
50 
51  Eigen::DiagonalMatrix<T,Eigen::Dynamic> D(K);
52  D.diagonal() = sds.inverse();
53  sds = sds.log(); // now unbounded
54 
55  Eigen::Matrix<T,Eigen::Dynamic,Eigen::Dynamic> R = D * Sigma * D;
56  // to hopefully prevent pivoting due to floating point error
57  R.diagonal().setOnes();
58  Eigen::LDLT<Eigen::Matrix<T,Eigen::Dynamic,Eigen::Dynamic> > ldlt;
59  ldlt = R.ldlt();
60  if( !ldlt.isPositive() ) return false;
61  Eigen::Matrix<T,Eigen::Dynamic,Eigen::Dynamic> U = ldlt.matrixU();
62 
63  size_t position = 0;
64  size_t pull = K - 1;
65 
66  Eigen::Array<T,Eigen::Dynamic,1> temp = U.row(0).tail(pull);
67  CPCs.head(pull) = temp;
68 
69  Eigen::Array<T,Eigen::Dynamic,1> acc(K);
70  acc(0) = -0.0;
71  acc.tail(pull) = 1.0 - temp.square();
72  for(size_t i = 1; i < (K - 1); i++) {
73  position += pull;
74  pull--;
75  temp = U.row(i).tail(pull);
76  temp /= sqrt(acc.tail(pull) / acc(i));
77  CPCs.segment(position, pull) = temp;
78  acc.tail(pull) *= 1.0 - temp.square();
79  }
80  CPCs = 0.5 * ( (1.0 + CPCs) / (1.0 - CPCs) ).log(); // now unbounded
81  return true;
82  }
